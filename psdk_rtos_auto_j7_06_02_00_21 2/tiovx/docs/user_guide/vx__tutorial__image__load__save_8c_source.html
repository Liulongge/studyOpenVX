233 vx_image create_image_from_file(vx_context context, char *filename, vx_bool convert_to_gray_scale)
234 {
235  vx_image image = NULL;
236  uint32_t width, height, stride;
237  vx_df_image df;
238  vx_status status;
239  void *data_ptr, *bmp_file_context;
240 
251  #ifdef SKIP_FILEIO
252  status = 0;
253  width = 640;
254  height = 480;
255  stride = 640;
256  if(convert_to_gray_scale)
257  df = (vx_df_image)VX_DF_IMAGE_U8;
258  else
259  df = (vx_df_image)VX_DF_IMAGE_RGB;
260  data_ptr = NULL;
261  bmp_file_context = NULL;
262  #else
263  status = bmp_file_read(

266  &width, &height, &stride, &df, &data_ptr,
267  &bmp_file_context);
268  #endif
269 
271  if(status==(vx_status)VX_SUCCESS)
272  {
286  image = vxCreateImage(context, width, height, df);
287  status = vxGetStatus((vx_reference)image);
289  if(status==(vx_status)VX_SUCCESS)
290  {
291 
313  vx_imagepatch_addressing_t image_addr;
314  vx_rectangle_t rect;
315  uint32_t bpp;
316 
317  if( df == (vx_df_image)VX_DF_IMAGE_U8 )
318  bpp = 1;
319  else
320  if( df == (vx_df_image)VX_DF_IMAGE_RGB )
321  bpp = 3;

324  bpp = 4;
325  else

327 
328  image_addr.dim_x = width;
329  image_addr.dim_y = height;
330  image_addr.stride_x = bpp;
331  image_addr.stride_y = stride;
332  image_addr.scale_x = VX_SCALE_UNITY;
333  image_addr.scale_y = VX_SCALE_UNITY;
334  image_addr.step_x = 1;
335  image_addr.step_y = 1;
336 
337  rect.start_x = 0;
338  rect.start_y = 0;
339  rect.end_x = width;
340  rect.end_y = height;
341 
342  if(data_ptr!=NULL)
343  {
344  vxCopyImagePatch(image,
345  &rect,
346  0,
347  &image_addr,
348  data_ptr,
349  (vx_enum)VX_WRITE_ONLY,
350  (vx_enum)VX_MEMORY_TYPE_HOST
351  );
352  }
354  }
362  bmp_file_read_release(bmp_file_context);
364  }
365  return image;
366 }

570 void ascii_file_read(char *filename, int num_elements, void* buffer, vx_enum data_type){
571  FILE* ptr_file;
572 
573  // Initialize buffer for reading each line of file
574  int nbytes = sizeof(float);
575  char* buff = malloc(nbytes);
576 
577  if (NULL == buff)
578  {
579  printf("FAILED ALLOCATING MEMORY\n");
580  return;
581  }
582 
583  // Initialize variables for tokenizing each line based on delimiter values
584  char* token;
585  char* delim = " \n\t,";
586  float val;
587 
588  // Initialize different data type arrays
589  vx_uint8* u8;
590  vx_int32* i32;
591  vx_float32* f32;
592  uint8_t val_u8;
593  int32_t val_int32;
594 
595  // Try reading in file
596  ptr_file = fopen(filename, "r");
597  if(!ptr_file){
598  printf("FAILED READING FILE\n");
599  free(buff);
600  return;
601  }
602 
603  // If file read is successful, try to populate matrix
604  int i = 0;
605  while(fgets(buff, nbytes, ptr_file) > 0){
606  // Split string into tokens at whitespaces, commas, newlines, or tabs
607  for(token = strtok(buff, delim); token != NULL; token = strtok(NULL, delim)){
608  val = strtod(token, NULL);
609 
610  switch(data_type){
611  case (vx_enum)VX_TYPE_FLOAT32:
612  f32 = buffer;
613  f32[i] = val;
614  break;
615 
616  case (vx_enum)VX_TYPE_UINT8:
617  u8 = buffer;
618  val_u8 = val;
619  u8[i] = val_u8;
620  break;
621 
622  case (vx_enum)VX_TYPE_INT32:
623  i32 = buffer;
624  val_int32 = val;
625  i32[i] = val_int32;
626  break;
627 
628  default:
629  break;
630  }
631 
632  i++;
633  if (i >= num_elements){
634  break;
635  }
636  }
637  // Need to break out of both loops when this condition is met
638  if (i >= num_elements){
639  break;
640  }
641  }
642  fclose(ptr_file);
643  free(buff);
644  //printf("val of first element: %d\n", u8[0]);
645 }
